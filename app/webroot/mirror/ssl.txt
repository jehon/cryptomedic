# Generate webserver.*
openssl genrsa -des3 -out webserver.key 2048
openssl req -new -key webserver.key -out webserver.csr

# Remove the password
mv webserver.key webserver.rien.key
openssl rsa -in webserver.rien.key -out webserver.key

# Generate server side certificate
openssl x509 -req -days 365 -in webserver.csr -signkey webserver.key -out webserver.crt

# Client certificate
openssl genrsa -out client.key 2048
openssl req -new -key client.key -out client.csr -config openssl.cnf
openssl x509 -req -days 365 -CA webserver.crt -CAkey webserver.key -CAcreateserial -in client.csr -out client.crt
openssl pkcs12 -export -clcerts -in client.crt -inkey client.key -out client.p12
